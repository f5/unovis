# XY Container + Stacked Bar + Tooltip + Crosshair

```tsx
// Import container, primitives and additional constants if needed
import { VisXYContainer, VisStackedBar, VisAxis, VisCrosshair, VisTooltip } from '@volterra/vis-react'

// Use them in your JSX
<VisXYContainer data={data} margin={{ top: 5, left: 5 }}>
  <VisStackedBar
    x={d => d.x}
    y={[d => d.y, d => d.y1, d => d.y2]}
    events={{
      [VisStackedBar.selectors.bar]: {
        click: d => console.log('clicked', d),
        mouseover: d => console.log('hovered', d),
      },
    }}
  />
  <VisAxis type='x' numTicks={5} tickFormat={(x) => `${x}ms`}/>
  <VisAxis type='y' tickFormat={(y) => `${y}bps`}/>
  <VisTooltip />
  <VisCrosshair template={d => `${d.y2.toFixed(2)}, ${d.y1.toFixed(2)}, ${d.y.toFixed(2)}`}/>
</VisXYContainer>
```
